// One-to-One (1–1)

Table person {
  id integer [primary key]
  name varchar
}

Table passport {
  id integer [primary key]
  passport_number varchar
  person_id integer [unique] // ensures 1:1
}

Ref: person.id < passport.person_id


// One-to-Many (1–N)
Table users {
  id integer [primary key]
  username varchar
}

Table posts {
  id integer [primary key]
  title varchar
  user_id integer [not null]
}

Ref: users.id < posts.user_id

// Many-to-Many (N–N)
Table students {
  id integer [primary key]
  name varchar
  email varchar
  created_at timestamp
}

Table courses {
  id integer [primary key]
  title varchar
  description text
  created_at timestamp
}

Table student_courses {
  student_id integer [not null]
  course_id integer [not null]
  enrollment_date timestamp
  indexes {
    (student_id, course_id) [unique] // composite primary key alternative
  }
}

// References
Ref: students.id < student_courses.student_id
Ref: courses.id < student_courses.course_id
